# 43 字符串相乘
## 思路
反转输入字符串，从头开始模拟乘法，最后将结果反转输出
## 出错的点
1. 输出结果不应该有前导零
2. 每一轮完整乘完后没有最后的向上进位
3. 最后向上进位后没有将进位归零
## code
```c++
class Solution {
public:
    void debug(vector<int> a){
        for(auto it = a.rbegin(); it != a.rend(); it++){
            cout<<*it;
        }
        cout<<endl;
    }
    string multiply(string num1, string num2) {
        reverse(num1.begin(), num1.end());
        reverse(num2.begin(), num2.end());
        if(num1.length() < num2.length()){
            swap(num1, num2);
        }
        int len1 = num1.length();
        int len2 = num2.length();
        vector<int> ansNum;
        int next = 0;
        //两重循环模拟乘法竖式
        //短的乘以长的
        for(int i = 0; i < len2; ++i){
            int a = num2[i] - '0';
            for(int j = 0; j < len1; ++j){
                int b = num1[j] - '0';
                int now = a * b + next;
                next = now / 10;
                now = now % 10;
                if(ansNum.size() <= i + j){
                    ansNum.push_back(now);
                } else {
                    ansNum[i + j] += now;
                    next += ansNum[i + j] / 10;
                    ansNum[i + j] %= 10;
                }
                // debug(ansNum);
            }
            if(next != 0){
                ansNum.push_back(next);
                next = 0;
            }
            // debug(ansNum);
        }
        
        stringstream ss;
        int flag = 1;
        for(auto it = ansNum.rbegin(); it != ansNum.rend(); it++){
            if(*it == 0 && flag){
                continue;
            }
            flag = 0;
            ss<<*it;
        }
        if(ss.str().length() == 0){
            return "0";
        }
        return ss.str();
    }
};
```
